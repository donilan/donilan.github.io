[bash] ##创建一个10M的空文件 $ dd if=/dev/zero of=/tmp/file.img bs=1M count=10 ##将一个循环设备与这个文件关联起来 $ losetup /dev/loop0 /tmp/file.img ##创建文件系统 $ mke2fs -c /dev/loop0 ##创建准备挂载的目录 $ mkdir /mnt/point ##挂载文件系统 $ mount -t ext2 /dev/loop0 /mnt/point [/bash]
