--- layout: post status: publish published: true title: Windows下批量重命名文件脚本 author: Doni author_login: D author_email: donilan.word@gmail.com wordpress_id: 171 wordpress_url: http://blog.sincerad.com/?p=171 date: '2011-08-24 02:29:22 +0800' date_gmt: '2011-08-24 02:29:22 +0800' categories: - IDE tags: [] comments: [] --- [bash] @echo off&setlocal EnableDelayedExpansion SET a=1 SET RENAME_FILE=jpg SET _date=%date: =0% SET _time=%time: =0% SET CURRENT_DATE=%_date:~0,4%_%_date:~5,2%_%_date:~8,2%_%_time:~0,2%%_time:~3,2% for /f "delims=" %%i in ('dir /b *.%RENAME_FILE%') do ( if not "%%~ni"=="%~n0" ( if !a! LSS 10 (ren "%%i" "%CURRENT_DATE%_00!a!.%RENAME_FILE%") else if !a! LSS 100 (ren "%%i" "%CURRENT_DATE%_0!a!.%RENAME_FILE%") else ren "%%i" "%CURRENT_DATE%_!a!.%RENAME_FILE%" set /a a+=1 ) ) [/bash]
