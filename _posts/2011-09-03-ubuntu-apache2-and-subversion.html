编译mod_dav_svn.so
[bash]
# 因为要获取一个模块需要很多源码..所以到临时文件里面处理
$ cd /tmp
# 获取源码
$ apt-get source subversion
# 解决依赖
$ sudo apt-get build-dep subversion
# 进到源码目录
$ cd subversion
# 生成配置文件
$ sudo ./configure --with-ssl --with-apxs=/usr/bin/apxs2
# 编译文件
$ sudo make
# 复制文件到apache2 modules下
$ cp subversion/mod_dav_svn/.libs/mod_dav_svn.so /usr/lib/apache2/modules/
[/bash]
